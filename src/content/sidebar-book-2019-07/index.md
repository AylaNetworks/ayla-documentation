  $('#delete-events-btn').click(function(event) {
    $('#events tr th input[type=checkbox]').prop('checked', false)
    let checkboxes = $('#events tbody tr td input[type=checkbox]:checked')
    $.each(checkboxes, function(index, checkbox) {

      $(tr1).remove()
      $(tr2).remove()
    })
  })
